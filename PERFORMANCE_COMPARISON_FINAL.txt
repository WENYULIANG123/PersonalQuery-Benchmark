  def search_batch(self, queries: List[str]) -> List[List[Tuple]]:
      # 批量编码查询
      query_embeddings = model.encode(["query: " + q for q in queries])
      # 批量计算相似度
      scores = util.cos_sim(query_embeddings, self.doc_embeddings)  # [N, 302k]
      # 批量 topk
      results = []
      for i in range(len(queries)):
          topk_values, topk_indices = torch.topk(scores[i], k=top_k)
          results.append([(self.doc_ids[idx], topk_values[j].item()) 
                          for j, idx in enumerate(topk_indices)])
      return results
